# WE WILL TRAIN OUR MODEL FOR 15 DIFFERENT VALUES OF K TO FIND BEST VALUE OF K

SSE = []

for x in range(1,15):
    KM_model = KMeans(n_clusters = x)
    KM_model.fit_predict(df[['Age','Income($)']])
    SSE.append(KM_model.inertia_)

print("All SSE are get calculated")
